# For Loop
stuffs = [ "Darth", "Luke", "Yoda", "Chewbacca", "Han" ]

for stuff in stuffs:
	print(stuff)
	
# While Loop
x = 0
y = 10

while x < y:
	print(str(x) + "/" + str(y))
	x += 1
	
print("We Made It!")
